🎮 HOW IT WORKS - DAMAGE FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

DAMAGE EVENT OCCURS
    │
    ├─→ player.take_damage(amount, source)
    │
    ├─→ Check: Shield > 0?
    │   ├─ YES: Remove min(amount, shield) from shield
    │   │        Calculate overflow: amount - shield_removed
    │   │
    │   └─ NO: Overflow = full amount
    │
    ├─→ Check: Overflow > 0 AND Hull > 0?
    │   ├─ YES: Reduce hull by overflow amount
    │   └─ NO: Hull stays same
    │
    ├─→ Emit: player_damaged(current_hull, current_shield)
    │
    └─→ Check: Hull <= 0?
        ├─ YES: Call die() → Emit player_died → queue_free()
        └─ NO: Continue normal gameplay

  Before:  Shield 30 | Hull 100
  Damage:  +50 (from enemy)
  Shield:  30 - 50 = 0 (absorbed 30)
  Overflow: 50 - 30 = 20
  Hull:    100 - 20 = 80
  Result:  Shield 0 | Hull 80 ✓

SCENARIO 3: Direct Hull Damage
  Before:  Shield 0 | Hull 100
  Damage:  +25 (from wall)
  Shield:  0 (already empty)
  Overflow: 25
  Hull:    100 - 25 = 75
  Result:  Shield 0 | Hull 75 ✓

SCENARIO 4: Player Death
  Before:  Shield 10 | Hull 30
  Damage:  +60 (from boss)
  Shield:  10 - 60 = 0 (absorbed 10)
  Overflow: 60 - 10 = 50
  Hull:    30 - 50 = -20 (capped at 0)
  Result:  Shield 0 | Hull 0
  Effect:  player_died signal emitted ✓
           queue_free() called
           Character deleted from scene

🔌 UI INTEGRATION EXAMPLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

extends Control

func _ready() -> void:
    var player: Player = get_tree().get_first_node_in_group("player")
    
    # Connect damage signal to update UI
    player.player_damaged.connect(_on_player_damaged)
    player.player_died.connect(_on_player_died)

func _on_player_damaged(hull: float, shield: float) -> void:
    # Update health bars
    $HullBar.value = hull
    $ShieldBar.value = shield
    print("Player: HP %d | Shield %d" % [int(hull), int(shield)])

func _on_player_died() -> void:
    # Show game over screen
    print("GAME OVER")
    get_tree().reload_current_scene()
